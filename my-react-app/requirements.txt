# Requirements / install instructions for `my-react-app`
# Use PowerShell on Windows. Run from the project root (this folder).

# 1) Basic install (recommended)
# Installs all dependencies and devDependencies listed in package.json
npm install

# 2) Explicit package list (optional)
# Dependencies
npm install react@^19.1.1 react-dom@^19.1.1 react-icons@^5.5.0 leaflet@^1.9.4 react-leaflet@^5.0.0 react-router-dom@^6.14.1 bcryptjs@^2.4.3

npm install -D @eslint/js@^9.36.0 @types/react@^19.1.16 @types/react-dom@^19.1.9 @vitejs/plugin-react@^5.0.4 cross-env@^10.1.0 electron@^39.0.0 eslint@^9.36.0 eslint-plugin-react-hooks@^5.2.0 eslint-plugin-react-refresh@^0.4.22 globals@^16.4.0 tailwindcss@^4.1.14 vite@^7.1.7

# 3) Run the app
# Dev server (vite)
npm run dev

# To start Electron in development (from this folder)
npm run electron:start
